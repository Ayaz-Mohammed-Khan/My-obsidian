class Solution {
    public boolean isHappy(int n) {
        int f = n;
        int s = n;
        do{
            f = helper(helper(f));
            s = helper(s);
            if(f == 1) return true;   
        }while(f != s);
        return false;
    }
    int helper(int n){
        int summ = 0;
        while(n != 0){
            int r = n % 10;
            n = n / 10;
            summ += r*r;
        }return summ;
    }
}
